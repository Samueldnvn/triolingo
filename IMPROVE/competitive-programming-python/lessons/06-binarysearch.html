What is Binary Search?

Binary search is a divide-and-conquer algorithm that finds the position of a target value within a sorted array. It compares the target to the middle element and eliminates half the remaining search space each iteration, achieving O(log n) time complexity.

Time Complexity

For array of size n:
Binary search: O(log n) comparisons
Linear search: O(n) comparisons

Example:
n = 1,000,000
Binary search: ~20 comparisons
Linear search: ~500,000 comparisons (average)
PREREQUISITE: Binary search requires sorted data in ascending or descending order.

Standard Binary Search

The fundamental algorithm

Iterative Implementation

def binary_search(nums, target):
    """Return index of target or -1 if not found"""
    left, right = 0, len(nums) - 1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if nums[mid] == target:
            return mid
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1

# Recursive version
def binary_search_recursive(nums, target, left=0, right=None):
    if right is None:
        right = len(nums) - 1
    
    if left > right:
        return -1
    
    mid = left + (right - left) // 2
    
    if nums[mid] == target:
        return mid
    elif nums[mid] < target:
        return binary_search_recursive(nums, target, mid + 1, right)
    else:
        return binary_search_recursive(nums, target, left, mid - 1)

# Example
nums = [1, 3, 5, 7, 9, 11, 13]
print(binary_search(nums, 7))  # 3

First/Last Occurrence

def find_first(nums, target):
    """Find first occurrence of target"""
    left, right = 0, len(nums) - 1
    result = -1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if nums[mid] == target:
            result = mid
            right = mid - 1  # Search left half
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return result

def find_last(nums, target):
    """Find last occurrence of target"""
    left, right = 0, len(nums) - 1
    result = -1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if nums[mid] == target:
            result = mid
            left = mid + 1  # Search right half
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return result

# Example
nums = [1, 2, 2, 2, 3, 4]
print(find_first(nums, 2))  # 1
print(find_last(nums, 2))   # 3

Using Python's bisect Module

import bisect

nums = [1, 3, 5, 7, 9]

# bisect_left: insertion point to maintain sorted order (leftmost)
pos = bisect.bisect_left(nums, 5)  # 2

# bisect_right: insertion point to maintain sorted order (rightmost)
pos = bisect.bisect_right(nums, 5)  # 3

# bisect: same as bisect_right
pos = bisect.bisect(nums, 5)  # 3

# insort: insert while maintaining order
bisect.insort(nums, 6)  # [1, 3, 5, 6, 7, 9]

# Check if exists
idx = bisect.bisect_left(nums, 5)
exists = idx < len(nums) and nums[idx] == 5

Search in Rotated Arrays

Binary search on rotated sorted arrays

Search in Rotated Sorted Array

def search_rotated(nums, target):
    """Search in rotated sorted array"""
    left, right = 0, len(nums) - 1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if nums[mid] == target:
            return mid
        
        # Determine which half is sorted
        if nums[left] <= nums[mid]:
            # Left half is sorted
            if nums[left] <= target < nums[mid]:
                right = mid - 1
            else:
                left = mid + 1
        else:
            # Right half is sorted
            if nums[mid] < target <= nums[right]:
                left = mid + 1
            else:
                right = mid - 1
    
    return -1

# Example: [4,5,6,7,0,1,2], target 0 -> 4

Find Minimum in Rotated Array

def find_min_rotated(nums):
    """Find minimum element in rotated sorted array"""
    left, right = 0, len(nums) - 1
    
    while left < right:
        mid = left + (right - left) // 2
        
        if nums[mid] > nums[right]:
            # Min is in right half
            left = mid + 1
        else:
            # Min is in left half (including mid)
            right = mid
    
    return nums[left]

# Example: [4,5,6,7,0,1,2] -> 0

Advanced Applications

Binary search beyond simple search

Search in 2D Matrix

def search_2d_matrix(matrix, target):
    """
    Search in 2D matrix where each row is sorted left to right,
    each column sorted top to bottom
    """
    if not matrix or not matrix[0]:
        return False
    
    m, n = len(matrix), len(matrix[0])
    row, col = 0, n - 1  # Start from top-right
    
    while row < m and col >= 0:
        if matrix[row][col] == target:
            return True
        elif matrix[row][col] < target:
            row += 1  # Move down
        else:
            col -= 1  # Move left
    
    return False

Binary Search on Answer

def sqrt_binary_search(x):
    """Find integer sqrt using binary search on answer"""
    if x < 2:
        return x
    
    left, right = 1, x // 2
    
    while left <= right:
        mid = left + (right - left) // 2
        sqrt = mid * mid
        
        if sqrt == x:
            return mid
        elif sqrt < x:
            left = mid + 1
        else:
            right = mid - 1
    
    return right

# Example: sqrt_binary_search(16) -> 4

Minimum Capacity to Ship Within D Days

def ship_within_days(weights, days):
    """Binary search on capacity"""
    left, right = max(weights), sum(weights)
    
    def can_ship(capacity):
        days_needed = 1
        current = 0
        for weight in weights:
            if current + weight > capacity:
                days_needed += 1
                current = weight
            else:
                current += weight
        return days_needed <= days
    
    while left < right:
        mid = left + (right - left) // 2
        if can_ship(mid):
            right = mid
        else:
            left = mid + 1
    
    return left
PATTERN: Binary search on answer works when the search space has a monotonic property where YES answers are contiguous.
