BFS and DFS

BFS - Level Order Traversal

from collections import deque

def bfs(graph, start):
    visited = set([start])
    queue = deque([start])
    result = []
    
    while queue:
        node = queue.popleft()
        result.append(node)
        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)
    
    return result

# Finds shortest path in unweighted graphs

DFS - Depth First Search

def dfs(graph, start, visited=None):
    if visited is None:
        visited = set()
    
    visited.add(start)
    result = [start]
    
    for neighbor in graph.get(start, []):
        if neighbor not in visited:
            result.extend(dfs(graph, neighbor, visited))
    
    return result

# Can be recursive or iterative using stack

Shortest Path Algorithms

Dijkstra's Algorithm

import heapq

def dijkstra(graph, start):
    """Shortest path from start to all nodes (weighted, non-negative)"""
    distances = {node: float('inf') for node in graph}
    distances[start] = 0
    pq = [(0, start)]
    visited = set()
    
    while pq:
        dist, node = heapq.heappop(pq)
        
        if node in visited:
            continue
        visited.add(node)
        
        for neighbor, weight in graph.get(node, {}).items():
            if neighbor not in visited:
                new_dist = dist + weight
                if new_dist < distances[neighbor]:
                    distances[neighbor] = new_dist
                    heapq.heappush(pq, (new_dist, neighbor))
    
    return distances

# Time: O((V+E)log V) with priority queue

Tree Algorithms

Is Graph a Tree?

def is_valid_tree(n, edges):
    """N nodes connected with N-1 edges and no cycle = tree"""
    if len(edges) != n - 1:
        return False
    
    graph = {i: [] for i in range(n)}
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)
    
    visited = set()
    
    def dfs(node, parent):
        visited.add(node)
        for neighbor in graph[node]:
            if neighbor != parent:
                if neighbor in visited:
                    return False
                if not dfs(neighbor, node):
                    return False
        return True
    
    return dfs(0, -1) and len(visited) == n

Minimum Spanning Tree

Kruskal's Algorithm with Union-Find

class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
    
    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        if self.rank[px] < self.rank[py]:
            self.parent[px] = py
        elif self.rank[px] > self.rank[py]:
            self.parent[py] = px
        else:
            self.parent[py] = px
            self.rank[px] += 1
        return True

def kruskal(n, edges):
    """Minimum spanning tree weight"""
    edges.sort(key=lambda x: x[2])
    uf = UnionFind(n)
    total = 0
    count = 0
    
    for u, v, w in edges:
        if uf.union(u, v):
            total += w
            count += 1
            if count == n - 1:
                break
    
    return total if count == n - 1 else None
